Ubuntu instuctions:
Install:

sudo apt-get update

sudo apt-get install mysql-server

sudo apt install mysql-client-core-5.7

sudo mysql_secure_installation utility

Setup DB:

sudo mysql -uroot -padmin

#Avoid needing to use sudo:
#From: https://stackoverflow.com/questions/37239970/connect-to-mysql-server-without-sudo
CREATE USER 'black'@'localhost' IDENTIFIED BY 'hack';
GRANT ALL PRIVILEGES ON ixhack.* TO 'black'@'localhost';
s
create database ixhack;

use ixhack;

CREATE TABLE `scholarship` (
  `link` varchar(255) NOT NULL,
  `school` varchar(100) DEFAULT NULL,
  `award_name` varchar(100) NOT NULL,
  `scholarship_provider` varchar(100) DEFAULT NULL,
  `amount` varchar(100) DEFAULT NULL,
  `value_description` varchar(500) DEFAULT NULL,
  `criteria_notes` varchar(500) DEFAULT NULL,
  `award_type` varchar(100) DEFAULT NULL,
  `deadline` varchar(100) DEFAULT NULL,
  `type_of_student` varchar(100) DEFAULT NULL,
  `level_of_study` varchar(100) DEFAULT NULL,
  `number_of_awards` mediumint(8) DEFAULT NULL,
  `renewable` bit(1) DEFAULT NULL,
  `field_of_study` varchar(100) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
